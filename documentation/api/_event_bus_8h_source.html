96 void unsubscribe(const Event& _type, HandlerId& _handlerId) {
97 if(!hasType(_type)) {
98 Util::Log::warn("Tried to unsubscribe Type ", typeid(_type).name(), " but it wasn't subscribed!");
99 return;
100 }
101
102 if (_handlerId.valid) {
103 handlers[_type].remove(_handlerId.id);
104 } else
105 Util::Log::warn("Tried to unsubscribe an event", typeid(_type).name()," that wasn't subscribed yet!");
106 }

108 bool dispatch(const Event& _type, AssociatedFunctionArgs... _args) {
109 if(!hasType(_type)) {
110 return false;
111 }
112
113 std::vector<bool> _results = handlers[_type](_args...);
114 return std::all_of(_results.begin(), _results.end(), [](bool v) { return v; });
115 }
116
117 bool isSubscribed(const Event& _type) {
118 return hasType(_type);
119 }

121 private:
